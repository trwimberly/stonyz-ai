    <script>
        // Global variables
        let currentUser = null;
        let users = [];
        
        // Bot responses
        const jokes = [
            "Why don't stoners ever win at hide and seek? Because they always get the munchies and have to come out for snacks! üçïüòÇ",
            "What do you call a stoner's favorite day? Weedsday! üòÇ",
            "What do you call a dinosaur that loves to chill? A dope-a-saurus! ü¶ïüòÇ"
        ];
        
        const horoscopes = [
            "The stars are aligning perfectly for you today! ‚ú® Your chill energy is attracting good vibes.",
            "Today's a great day to trust your instincts and go with the flow! üåä",
            "The universe is sending you peaceful energy today. Perfect time to relax and enjoy! üåü"
        ];
        
        const vibes = [
            "Your vibe is absolutely immaculate today! ‚ú® Perfect energy for exploring new horizons!",
            "I'm sensing some serious chill energy from you! üåä That's exactly the vibe we're here for!",
            "Your aura is glowing today! üåü Time to embrace whatever the universe has in store!"
        ];

        // Tab switching functions
        function showSignIn() {
            document.getElementById('signInSection').classList.add('active');
            document.getElementById('signUpSection').classList.remove('active');
            document.querySelectorAll('.tab')[0].classList.add('active');
            document.querySelectorAll('.tab')[1].classList.remove('active');
            clearAlert();
        }
        
        function showSignUp() {
            document.getElementById('signUpSection').classList.add('active');
            document.getElementById('signInSection').classList.remove('active');
            document.querySelectorAll('.tab')[1].classList.add('active');
            document.querySelectorAll('.tab')[0].classList.remove('active');
            clearAlert();
        }

        // Alert functions
        function showAlert(message, type = 'error') {
            const alertContainer = document.getElementById('alertContainer');
            alertContainer.innerHTML = `<div class="alert ${type}">${message}</div>`;
        }

        function clearAlert() {
            document.getElementById('alertContainer').innerHTML = '';
        }

        // Authentication functions
        function signUp(event) {
            event.preventDefault();
            
            const name = document.getElementById('signUpName').value.trim();
            const username = document.getElementById('signUpUsername').value.trim();
            const password = document.getElementById('signUpPassword').value;
            const confirmPassword = document.getElementById('confirmPassword').value;
            
            if (!name || !username || !password) {
                showAlert('Please fill in all fields');
                return;
            }
            
            if (password !== confirmPassword) {
                showAlert('Passwords do not match');
                return;
            }
            
            if (password.length < 6) {
                showAlert('Password must be at least 6 characters');
                return;
            }
            
            if (users.find(user => user.username === username)) {
                showAlert('Username already exists');
                return;
            }
            
            const newUser = {
                id: Date.now(),
                name: name,
                username: username,
                password: password
            };
            
            users.push(newUser);
            currentUser = newUser;
            
            showChatInterface();
        }

        function signIn(event) {
            event.preventDefault();
            
            const username = document.getElementById('signInUsername').value.trim();
            const password = document.getElementById('signInPassword').value;
            
            if (!username || !password) {
                showAlert('Please enter username and password');
                return;
            }
            
            const user = users.find(u => u.username === username && u.password === password);
            
            if (!user) {
                showAlert('Invalid username or password');
                return;
            }
            
            currentUser = user;
            showChatInterface();
        }

        function logout() {
            currentUser = null;
            document.getElementById('authSection').style.display = 'block';
            document.getElementById('chatSection').classList.remove('active');
            document.getElementById('signInUsername').value = '';
            document.getElementById('signInPassword').value = '';
            clearAlert();
        }

        function showChatInterface() {
            document.getElementById('authSection').style.display = 'none';
            document.getElementById('chatSection').classList.add('active');
            document.getElementById('currentUserName').textContent = currentUser.name;
            
            // Clear messages and add welcome message
            const messagesDiv = document.getElementById('messages');
            messagesDiv.innerHTML = '';
            addMessage(`Hey ${currentUser.name}! üåø Welcome to Stonyz.ai! I'm here to chat, tell jokes, share horoscopes, and give you good vibes. What's on your mind today?`, 'bot');
        }

        // Chat functions
        function addMessage(text, sender) {
            const messagesDiv = document.getElementById('messages');
            const messageDiv = document.createElement('div');
            messageDiv.className = `message ${sender}`;
            messageDiv.textContent = text;
            messagesDiv.appendChild(messageDiv);
            messagesDiv.scrollTop = messagesDiv.scrollHeight;
        }

        function sendMessage() {
            const input = document.getElementById('messageInput');
            const message = input.value.trim();
            
            if (!message) return;
            
            addMessage(message, 'user');
            input.value = '';
            
            // Bot response
            setTimeout(() => {
                const response = generateResponse(message);
                addMessage(response, 'bot');
            }, 500);
        }

        function generateResponse(message) {
            const input = message.toLowerCase();
            const userName = currentUser.name;
            
            if (input.includes('hi') || input.includes('hello') || input.includes('hey')) {
                return `Hey there, ${userName}! üåø What's bringing you to the good vibes today?`;
            } else if (input.includes('product') || input.includes('buy') || input.includes('shop')) {
                return `Dude, ${userName}, we've got some absolutely righteous products! üî• Everything from premium CBD to the dopest apparel!`;
            } else {
                return `That's totally cool, ${userName}! Tell me more about what you're looking for and I'll point you in the right direction! üåø`;
            }
        }

        function handleKeyPress(event) {
            if (event.key === 'Enter') {
                sendMessage();
            }
        }

        // Feature functions
        function tellJoke() {
            const randomJoke = jokes[Math.floor(Math.random() * jokes.length)];
            addMessage(randomJoke, 'bot');
        }

        function getHoroscope() {
            const randomHoroscope = horoscopes[Math.floor(Math.random() * horoscopes.length)];
            addMessage(`${currentUser.name}, here's your cosmic reading: ${randomHoroscope}`, 'bot');
        }

        function vibeCheck() {
            const randomVibe = vibes[Math.floor(Math.random() * vibes.length)];
            addMessage(randomVibe.replace('Your', `${currentUser.name}, your`), 'bot');
        }

        // Initialize
        console.log('üåø Stonyz.ai loaded successfully!');
    </script>
